```go
func (s *transactionService) maybeDispatchSplitUpdatedNotification(ctx context.Context, callerUserID int, sourceIDs []int, data *transactionUpdateData) {
    var events []domain.NotificationEvent
    switch {
    case data.scenario.AddedSplit():
        if callerUserID == lo.FromPtr(data.previousTransaction.OriginalUserID) { // D-03 self-edit guard
            return
        }
        for _, ss := range data.req.SplitSettings {
            if ss.UserConnection == nil { continue }
            recipientID := ss.UserConnection.ToUserID
            if recipientID == callerUserID { recipientID = ss.UserConnection.FromUserID }
            if recipientID == callerUserID { continue }
            events = append(events, domain.NotificationEvent{
                RecipientUserID: recipientID, ActorUserID: callerUserID,
                Type: domain.NotificationTypeSplitUpdated, EntityType: "transaction",
                EntityID: data.previousTransaction.ID, Amount: lo.FromPtr(data.req.Amount),
            })
        }
    case data.scenario.RemovedSplit():
        if callerUserID == lo.FromPtr(data.previousTransaction.OriginalUserID) { return } // D-03; D-04 still notifies recipient
        for _, lt := range data.previousTransaction.LinkedTransactions {
            if lt.UserID == callerUserID { continue }
            events = append(events, domain.NotificationEvent{
                RecipientUserID: lt.UserID, ActorUserID: callerUserID,
                Type: domain.NotificationTypeSplitUpdated, EntityType: "transaction",
                EntityID: lt.ID, Amount: lt.Amount, // deep-link to soft-deleted linked tx (D-04)
            })
        }
    case data.isLinkedTxEdit && data.req.Amount != nil && *data.req.Amount > 0:
        // caller IS the partner editing their linked side; recipient = source tx owner
        for _, srcID := range sourceIDs {
            srcTx, err := s.getByID(ctx, callerUserID, srcID)
            if err != nil || srcTx == nil || srcTx.UserID == callerUserID { continue }
            events = append(events, domain.NotificationEvent{
                RecipientUserID: srcTx.UserID, ActorUserID: callerUserID,
                Type: domain.NotificationTypeSplitUpdated, EntityType: "transaction",
                EntityID: srcID, Amount: *data.req.Amount,
            })
        }
    }
    if len(events) > 0 {
        go s.services.Notification.Dispatch(context.Background(), events)
    }
}
```

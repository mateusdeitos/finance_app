    ```ts
    import { Transactions } from "@/types/transactions";
    import type { TransactionFormValues } from "./transactionFormSchema";

    interface Refs {
      accounts: Transactions.Account[];
      categories: Transactions.Category[];
      tags: Transactions.Tag[];
    }

    /**
     * Maps a template payload onto transaction-form fields, clearing references
     * (account/category/tag) that no longer exist so a stale template applies
     * cleanly (APPLY-04). Never sets amount or date. Returns a partial patch;
     * the caller applies it via setValue and blanks the amount separately.
     */
    export function buildTemplateFormPatch(
      payload: Transactions.TemplatePayload,
      { accounts, categories, tags }: Refs,
    ): Partial<TransactionFormValues> {
      const accountExists = (id?: number | null) =>
        id != null && accounts.some((a) => a.id === id);
      const categoryExists = (id?: number | null) =>
        id != null && categories.some((c) => c.id === id);

      const tagNames = (payload.tag_ids ?? [])
        .map((id) => tags.find((t) => t.id === id)?.name)
        .filter((n): n is string => Boolean(n));

      return {
        transaction_type: payload.type,
        description: payload.description ?? "",
        account_id: accountExists(payload.account_id) ? payload.account_id! : undefined,
        category_id: categoryExists(payload.category_id) ? payload.category_id! : null,
        destination_account_id: accountExists(payload.destination_account_id)
          ? payload.destination_account_id!
          : null,
        tags: tagNames,
        split_settings: payload.split_settings ?? [],
      };
    }
    ```
    (Match the exact optional/nullable field types from `TransactionFormValues` — if `account_id` is `number | undefined` vs `number | null` in the schema, use the schema's type so `tsc` passes. `categories` here is the FLATTENED category list used by the form, `useFlattenCategories` output; the executor should confirm its element type has an `id`. No `any`.)
